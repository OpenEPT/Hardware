141drv_system_status_t DRV_SYSTEM_InitCoreFunc()
142{
143 prvDRV_SYSTEM_CACHE_Enable();
144 if(prvDRV_SYSTEM_CPU2_Wait(0xFFFF) != DRV_SYSTEM_STATUS_OK) return DRV_SYSTEM_STATUS_ERROR;
145 if(HAL_Init() != HAL_OK) return DRV_SYSTEM_STATUS_ERROR;
146 if(prvDRV_SYSTEM_MPU_Init() != DRV_SYSTEM_STATUS_OK) return DRV_SYSTEM_STATUS_ERROR;
147 if(prvDRV_SYSTEM_CLOCK_Init() != DRV_SYSTEM_STATUS_OK) return DRV_SYSTEM_STATUS_ERROR;
148
149 __HAL_RCC_HSEM_CLK_ENABLE();
150 /*Take HSEM */
151 HAL_HSEM_FastTake(HSEM_ID_0);
152 /*Release HSEM in order to notify the CPU2(CM4)*/
153 HAL_HSEM_Release(HSEM_ID_0,0);
154
155
156 if(prvDRV_SYSTEM_CPU2_Wait(0xFFFF) != DRV_SYSTEM_STATUS_OK) return DRV_SYSTEM_STATUS_ERROR;
157
158
159 return DRV_SYSTEM_STATUS_OK;
160}

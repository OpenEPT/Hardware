83drv_uart_status_t DRV_UART_Instance_Init(drv_uart_instance_t instance, drv_uart_config_t* config)
84{
85 if(prvDRV_UART_INSTANCES[instance].lock != NULL) return DRV_UART_STATUS_ERROR;
86
87 prvDRV_UART_INSTANCES[instance].lock = xSemaphoreCreateMutex();
88
89 if(prvDRV_UART_INSTANCES[instance].lock == NULL) return DRV_UART_STATUS_ERROR;
90
91 switch(instance)
92 {
93 case DRV_UART_INSTANCE_1:
94 prvDRV_UART_INSTANCES[instance].deviceHandler.Instance = USART1;
95 break;
96 case DRV_UART_INSTANCE_3:
97 prvDRV_UART_INSTANCES[instance].deviceHandler.Instance = USART3;
98 break;
99
100 }
101
102 /*TODO: Only baudrate is configurable*/
103 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.BaudRate = config->baudRate;
104 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.WordLength = UART_WORDLENGTH_8B;
105 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.StopBits = UART_STOPBITS_1;
106 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.Parity = UART_PARITY_NONE;
107 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.Mode = UART_MODE_TX_RX;
108 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.HwFlowCtl = UART_HWCONTROL_NONE;
109 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.OverSampling = UART_OVERSAMPLING_16;
110 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.OneBitSampling = UART_ONE_BIT_SAMPLE_DISABLE;
111 prvDRV_UART_INSTANCES[instance].deviceHandler.Init.ClockPrescaler = UART_PRESCALER_DIV1;
112 prvDRV_UART_INSTANCES[instance].deviceHandler.AdvancedInit.AdvFeatureInit = UART_ADVFEATURE_NO_INIT;
113
114 if (HAL_UART_Init(&prvDRV_UART_INSTANCES[instance].deviceHandler) != HAL_OK) return DRV_UART_STATUS_ERROR;
115 if (HAL_UARTEx_SetTxFifoThreshold(&prvDRV_UART_INSTANCES[instance].deviceHandler, UART_TXFIFO_THRESHOLD_1_8) != HAL_OK) return DRV_UART_STATUS_ERROR;
116 if (HAL_UARTEx_SetRxFifoThreshold(&prvDRV_UART_INSTANCES[instance].deviceHandler, UART_RXFIFO_THRESHOLD_1_8) != HAL_OK) return DRV_UART_STATUS_ERROR;
117 if (HAL_UARTEx_DisableFifoMode(&prvDRV_UART_INSTANCES[instance].deviceHandler) != HAL_OK) return DRV_UART_STATUS_ERROR;
118
119 prvDRV_UART_INSTANCES[instance].initState = DRV_UART_INITIALIZATION_STATUS_INIT;
120
121 return DRV_UART_STATUS_OK;
122}
123drv_uart_status_t DRV_UART_TransferData(drv_uart_instance_t instance, uint8_t* buffer, uint8_t size, uint32_t timeout)
124{
125 if(prvDRV_UART_INSTANCES[instance].initState != DRV_UART_INITIALIZATION_STATUS_INIT || prvDRV_UART_INSTANCES[instance].lock == NULL) return DRV_UART_STATUS_ERROR;
126
127 if(xSemaphoreTake(prvDRV_UART_INSTANCES[instance].lock, pdMS_TO_TICKS(timeout)) != pdTRUE) return DRV_UART_STATUS_ERROR;
128
129 if(HAL_UART_Transmit(&prvDRV_UART_INSTANCES[instance].deviceHandler, buffer, size, timeout) != HAL_OK) return DRV_UART_STATUS_ERROR;
130
131 if(xSemaphoreGive(prvDRV_UART_INSTANCES[instance].lock) != pdTRUE) return DRV_UART_STATUS_ERROR;
132
133 return DRV_UART_STATUS_OK;
134}

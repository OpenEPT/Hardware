92static void prvNETWORK_Task()
93{
94 memset(&prvNETWORK_MAC_CONFIG, 0, sizeof(ETH_MACConfigTypeDef));
95 int32_t PHYLinkState = 0;
96 uint32_t linkchanged = 0U, speed = 0U, duplex = 0U;
97 prvNETWORK_DATA.linkStatus = SYSTEM_LINK_STATUS_DOWN;
98 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, "Network service started\r\n");
99 for(;;)
100 {
101 switch(prvNETWORK_DATA.state)
102 {
103 case NETWORK_STATE_INIT:
104 /* Initilialize the LwIP stack with RTOS */
105 tcpip_init( NULL, NULL );
106
107 /* IP addresses initialization without DHCP (IPv4) */
108 inet_pton(AF_INET, NETWORK_DEVICE_IP_ADDRESS, &prvNETWORK_DATA.ipaddr);
109 inet_pton(AF_INET, NETWORK_DEVICE_IP_MASK, &prvNETWORK_DATA.netmask);
110 inet_pton(AF_INET, NETWORK_DEVICE_IP_GW, &prvNETWORK_DATA.gw);
111
112 /* add the network interface (IPv4/IPv6) with RTOS */
113 netif_add(&prvNETWORK_DATA.gnetif, &prvNETWORK_DATA.ipaddr, &prvNETWORK_DATA.netmask, &prvNETWORK_DATA.gw, NULL, &ethernetif_init, &tcpip_input);
114
115 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, "Network interface added - Info:\r\n");
116 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - IP : %s\r\n", NETWORK_DEVICE_IP_ADDRESS);
117 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - MASK : %s\r\n", NETWORK_DEVICE_IP_MASK);
118 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - Gateway : %s\r\n", NETWORK_DEVICE_IP_GW);
119
120 /* Registers the default network interface */
121 netif_set_default(&prvNETWORK_DATA.gnetif);
122
123 if (netif_is_link_up(&prvNETWORK_DATA.gnetif))
124 {
125 /* When the netif is fully configured this function must be called */
126 netif_set_up(&prvNETWORK_DATA.gnetif);
127 SYSTEM_SetLinkStatus(SYSTEM_LINK_STATUS_UP);
128 prvNETWORK_DATA.linkStatus = SYSTEM_LINK_STATUS_UP;
129 LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, "Network interface up\r\n");
130 }
131 else
132 {
133 /* When the netif link is down this function must be called */
134 netif_set_down(&prvNETWORK_DATA.gnetif);
135 SYSTEM_SetLinkStatus(SYSTEM_LINK_STATUS_DOWN);
136 prvNETWORK_DATA.linkStatus = SYSTEM_LINK_STATUS_DOWN;
137 LOGGING_Write("Network", LOGGING_MSG_TYPE_WARNNING, "Network interface down\r\n");
138 }
139
140 /* Set the link callback function, this function is called on change of link status*/
141 netif_set_link_callback(&prvNETWORK_DATA.gnetif, prvNETWORK_LinkStatusUpdated);
142
143
144 lwiperf_start_tcp_server_default(NULL, NULL);
145 xSemaphoreGive(prvNETWORK_DATA.initSig);
146 prvNETWORK_DATA.state = NETWORK_STATE_SERVICE;
147 break;
148 case NETWORK_STATE_SERVICE:
149 PHYLinkState = LAN8742_GetLinkState(&LAN8742);
150
151 if(netif_is_link_up(&prvNETWORK_DATA.gnetif) && (PHYLinkState <= LAN8742_STATUS_LINK_DOWN))
152 {
153 HAL_ETH_Stop_IT(&HETH);
154 netif_set_down(&prvNETWORK_DATA.gnetif);
155 netif_set_link_down(&prvNETWORK_DATA.gnetif);
156 }
157 else if(!netif_is_link_up(&prvNETWORK_DATA.gnetif) && (PHYLinkState > LAN8742_STATUS_LINK_DOWN))
158 {
159 switch (PHYLinkState)
160 {
161 case LAN8742_STATUS_100MBITS_FULLDUPLEX:
162 duplex = ETH_FULLDUPLEX_MODE;
163 speed = ETH_SPEED_100M;
164 linkchanged = 1;
165 break;
166 case LAN8742_STATUS_100MBITS_HALFDUPLEX:
167 duplex = ETH_HALFDUPLEX_MODE;
168 speed = ETH_SPEED_100M;
169 linkchanged = 1;
170 break;
171 case LAN8742_STATUS_10MBITS_FULLDUPLEX:
172 duplex = ETH_FULLDUPLEX_MODE;
173 speed = ETH_SPEED_10M;
174 linkchanged = 1;
175 break;
176 case LAN8742_STATUS_10MBITS_HALFDUPLEX:
177 duplex = ETH_HALFDUPLEX_MODE;
178 speed = ETH_SPEED_10M;
179 linkchanged = 1;
180 break;
181 default:
182 break;
183 }
184
185 if(linkchanged)
186 {
187 /* Get MAC Config MAC */
188 HAL_ETH_GetMACConfig(&HETH, &prvNETWORK_MAC_CONFIG);
189 prvNETWORK_MAC_CONFIG.DuplexMode = duplex;
190 prvNETWORK_MAC_CONFIG.Speed = speed;
191 HAL_ETH_SetMACConfig(&HETH, &prvNETWORK_MAC_CONFIG);
192 HAL_ETH_Start_IT(&HETH);
193 netif_set_up(&prvNETWORK_DATA.gnetif);
194 netif_set_link_up(&prvNETWORK_DATA.gnetif);
195 }
196 }
197 vTaskDelay(pdMS_TO_TICKS(100));
198 break;
199 case NETWORK_STATE_ERROR:
200 SYSTEM_ReportError(SYSTEM_ERROR_LEVEL_LOW);
201 vTaskDelay(portMAX_DELAY);
202 break;
203 }
204
205 }
206}

211network_status_t NETWORK_Init(uint32_t timeout)
212{
213 if(xTaskCreate(prvNETWORK_Task,
214 NETWORK_TASK_NAME,
215 NETWORK_TASK_STACK_SIZE,
216 NULL,
217 NETWORK_TASK_PRIO,
218 &prvNETWORK_TASK_HANDLE) != pdTRUE) return NETWORK_STATUS_ERROR;
219
220 prvNETWORK_DATA.initSig = xSemaphoreCreateBinary();
221
222 if(prvNETWORK_DATA.initSig == NULL) return NETWORK_STATUS_ERROR;
223
224 prvNETWORK_DATA.state = NETWORK_STATE_INIT;
225
226 if(xSemaphoreTake(prvNETWORK_DATA.initSig, pdMS_TO_TICKS(timeout)) != pdFALSE) return NETWORK_STATUS_ERROR;
227
228 return NETWORK_STATUS_OK;
229}
